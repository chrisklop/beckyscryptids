```ts
/**
 * Largest N such that computeBulkCost(base, growth, owned, N) <= available.
 *
 * Closed-form via inverse of the geometric-series formula:
 *   cost(N) = base × growth^owned × (growth^N - 1) / (growth - 1) ≤ available
 *   → N ≤ log( 1 + available × (growth - 1) / (base × growth^owned) ) / log(growth)
 *
 * Returns 0 when the player can't even afford one (i.e. available < cost(owned)).
 */
export function maxAffordableBulk(
  base: number,
  growth: number,
  owned: number,
  available: number,
): number {
  if (available <= 0) return 0;
  const costOfNext = base * Math.pow(growth, owned);
  if (available < costOfNext) return 0;
  const numerator = Math.log(1 + (available * (growth - 1)) / costOfNext);
  const denominator = Math.log(growth);
  const n = Math.floor(numerator / denominator);
  // Floating-point guard: the floor can land one off due to rounding in the log.
  // Verify with the forward formula and step down if it overshoots.
  if (computeBulkCost(base, growth, owned, n) > available + 1e-9) {
    return Math.max(0, n - 1);
  }
  return n;
}
```

```ts
/**
 * Pecorella overtake heuristic — pick the generator with the lowest
 *   cost / nps + cost / (nps + delta_rate_from_buying_one)
 * score. Lower is better (means faster total-time to afford it AND the
 * next opportunity it unlocks).
 *
 * Click-driven generators that are below auto_unlock_at contribute 0 to
 * rate-after-buy, so they only become attractive once close to unlock.
 */
export function pickOptimalGenerator(
  candidates: readonly GeneratorTier[],
  state: { rumor: number; ownedByGenerator: Record<string, number> },
  globalMultiplier: number,
): string | null {
  if (candidates.length === 0) return null;

  // Total current nps across all candidates.
  let nps = 0;
  for (const g of candidates) {
    const owned = state.ownedByGenerator[g.id] ?? 0;
    nps += generatorProduction(g, owned, globalMultiplier);
  }
  // Fallback: if nps is 0, we'd divide by 0. Treat as a small constant so
  // the heuristic still ranks cheapest-first.
  const safeNps = nps > 0 ? nps : 1;

  let bestId: string | null = null;
  let bestScore = Infinity;

  for (const g of candidates) {
    const owned = state.ownedByGenerator[g.id] ?? 0;
    const cost = computeCost(g.base_cost, g.cost_growth, owned);

    // rate gained by buying one more
    const before = generatorProduction(g, owned, globalMultiplier);
    const after = generatorProduction(g, owned + 1, globalMultiplier);
    const delta = Math.max(0, after - before);

    const score = cost / safeNps + cost / (safeNps + delta);
    if (score < bestScore) {
      bestScore = score;
      bestId = g.id;
    }
  }

  return bestId;
}
```

```ts
import LZString from 'lz-string';

export const CURRENT_SAVE_VERSION = 2;

export type BulkBuyMultiplier = 1 | 10 | 100 | 'max';

export interface SaveState {
  version: number;
  current_era: string;
  rumor: number;
  lifetime_rumor: number;
  memetic_inheritance: number;
  owned_by_generator: Record<string, number>;
  unlocked_codex: string[];
  saved_at_ms: number;
  // v2 additions
  prestige_count: number;
  seen_toast_events: string[];
  bulk_buy_multiplier: BulkBuyMultiplier;
  show_best_buy_hint: boolean;
}

const V2_DEFAULTS = {
  prestige_count: 0,
  seen_toast_events: [] as string[],
  bulk_buy_multiplier: 1 as BulkBuyMultiplier,
  show_best_buy_hint: true,
};

/**
 * Migrate a save object of any prior version to the current shape.
 * Idempotent: calling on an already-v2 save returns it unchanged.
 */
export function migrateSave(raw: any): SaveState {
  let s = { ...raw };
  if ((s.version ?? 1) < 2) {
    s = {
      ...s,
      ...V2_DEFAULTS,
      // Preserve any pre-existing v2 fields the caller passed (defensive).
      prestige_count: s.prestige_count ?? V2_DEFAULTS.prestige_count,
      seen_toast_events: s.seen_toast_events ?? V2_DEFAULTS.seen_toast_events,
      bulk_buy_multiplier: s.bulk_buy_multiplier ?? V2_DEFAULTS.bulk_buy_multiplier,
      show_best_buy_hint: s.show_best_buy_hint ?? V2_DEFAULTS.show_best_buy_hint,
      version: 2,
    };
  }
  return s as SaveState;
}

export function serializeSave(state: SaveState): string {
  const json = JSON.stringify(state);
  return LZString.compressToUTF16(json);
}

export function deserializeSave(encoded: string): SaveState {
  const json = LZString.decompressFromUTF16(encoded);
  if (!json) throw new Error('Failed to decompress save');
  const parsed = JSON.parse(json);
  if (typeof parsed !== 'object' || !parsed.version) {
    throw new Error('Save missing version field');
  }
  return migrateSave(parsed);
}

const STORAGE_KEY = 'playbook.save';

export function writeLocalSave(state: SaveState): void {
  localStorage.setItem(STORAGE_KEY, serializeSave(state));
}

export function readLocalSave(): SaveState | null {
  const raw = localStorage.getItem(STORAGE_KEY);
  if (!raw) return null;
  try {
    return deserializeSave(raw);
  } catch (err) {
    console.error('Save corrupted, ignoring:', err);
    return null;
  }
}
```

Create `src/ui/toast-state.ts`:

```ts
import { reactive, computed } from 'vue';
import { state } from './state';

export interface ActiveToast {
  id: string;            // event id (e.g. 'auto-unlock:antiquity:spread-rumor')
  message: string;
  era_id: string;        // for theme-aware styling
  fired_at_ms: number;
}

const MAX_VISIBLE = 3;
const DISMISS_AFTER_MS = 3000;

const queue = reactive<{ active: ActiveToast[] }>({ active: [] });

export const activeToasts = computed(() => queue.active);

/**
 * Fire a toast iff its id has not been seen before. Add the id to the seen set
 * (which is persisted) so it won't re-fire on reload.
 */
export function fireToast(opts: { id: string; message: string; era_id: string }): void {
  if (state.seenToastEvents.has(opts.id)) return;
  state.seenToastEvents.add(opts.id);

  const toast: ActiveToast = {
    id: opts.id,
    message: opts.message,
    era_id: opts.era_id,
    fired_at_ms: Date.now(),
  };
  queue.active.push(toast);
  // Cap visible by dropping the oldest (slide-out is the component's concern).
  while (queue.active.length > MAX_VISIBLE) {
    queue.active.shift();
  }
  // Auto-dismiss after timeout.
  if (typeof window !== 'undefined') {
    window.setTimeout(() => dismissToast(opts.id), DISMISS_AFTER_MS);
  }
}

export function dismissToast(id: string): void {
  const idx = queue.active.findIndex(t => t.id === id);
  if (idx >= 0) queue.active.splice(idx, 1);
}
```

```ts
import { computed } from 'vue';
import { state, buyGenerator, click } from '../state';
import { computeCost } from '../../game/era-layer';
import type { GeneratorTier } from '../../content/schema';

const props = defineProps<{ gen: GeneratorTier }>();

const owned = computed(() => state.ownedByGenerator[props.gen.id] ?? 0);
const cost = computed(() => computeCost(props.gen.base_cost, props.gen.cost_growth, owned.value));
const canAfford = computed(() => state.rumor >= cost.value);

/** The next-milestone owned threshold, or null if past the final milestone. */
const nextMilestone = computed<number | null>(() => {
  for (const m of props.gen.milestones) {
    if (owned.value < m) return m;
  }
  return null;
});

/** The previous milestone (defaults to 0 if we haven't passed any). */
const prevMilestone = computed<number>(() => {
  let prev = 0;
  for (const m of props.gen.milestones) {
    if (owned.value >= m) prev = m;
    else break;
  }
  return prev;
});

/** Current multiplier from milestones passed (each pass doubles). */
const currentMilestoneMult = computed<number>(() => {
  let mult = 1;
  for (const m of props.gen.milestones) {
    if (owned.value >= m) mult *= 2;
  }
  return mult;
});

/** Multiplier after the NEXT milestone is hit. */
const nextMilestoneMult = computed<number>(() => currentMilestoneMult.value * 2);

/** Progress from previous milestone to next, 0..1. */
const milestoneProgress = computed<number>(() => {
  if (nextMilestone.value === null) return 1;
  const span = nextMilestone.value - prevMilestone.value;
  const into = owned.value - prevMilestone.value;
  return span > 0 ? Math.min(1, Math.max(0, into / span)) : 0;
});

function tap() {
  if (props.gen.is_click_driven) {
    click();
    if (state.rumor >= cost.value) {
      buyGenerator(props.gen.id);
    }
  } else {
    buyGenerator(props.gen.id);
  }
}

function formatCost(n: number): string {
  if (n < 1000) return n.toFixed(0);
  if (n < 1e6) return (n / 1000).toFixed(1) + 'K';
  if (n < 1e9) return (n / 1e6).toFixed(1) + 'M';
  return n.toExponential(1);
}
```

```ts
import { computeBulkCost, maxAffordableBulk } from '../game/era-layer';

/**
 * Buy N units of a generator. Returns the number of units actually purchased
 * (may be less than `n` if the player can't afford `n`, but is greedy-best-effort).
 *
 * When `n === 'max'`, buys as many as the player can afford using maxAffordableBulk.
 */
export function buyGenerator(genId: string, n: number | 'max' = 1): number {
  const gen = currentEra.value.generators.find(g => g.id === genId);
  if (!gen) return 0;
  const owned = state.ownedByGenerator[genId] ?? 0;

  let buyN: number;
  if (n === 'max') {
    buyN = maxAffordableBulk(gen.base_cost, gen.cost_growth, owned, state.rumor);
  } else {
    const fullCost = computeBulkCost(gen.base_cost, gen.cost_growth, owned, n);
    if (state.rumor >= fullCost) {
      buyN = n;
    } else {
      // Partial buy: as many as the player can afford.
      buyN = maxAffordableBulk(gen.base_cost, gen.cost_growth, owned, state.rumor);
    }
  }

  if (buyN <= 0) return 0;
  const totalCost = computeBulkCost(gen.base_cost, gen.cost_growth, owned, buyN);
  state.rumor -= totalCost;
  state.ownedByGenerator[genId] = owned + buyN;
  return buyN;
}
```

```ts
export const nextHiddenGenerator = computed(() => {
```
